Server.py:
from flask import Flask


class Server:
    def __init__(self):
        self.app = Flask(__name__)

        self.register_routes()

    def register_routes(self):
        @self.app.route('/', methods=["GET"])
        def home():
            return "Hello World"

main.py:
from Server import Server

if __name__ == '__main__':
    server = Server()
    server.app.run()
